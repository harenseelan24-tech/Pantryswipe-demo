### Parsing — IMPORTANT:
Wrap JSON.parse in a robust helper. Add this function to aiChef.ts:

  function parseClaudeJSON(raw: string): T {
    // Strip markdown fences Claude sometimes adds despite instructions
    const cleaned = raw
      .replace(/^```(?:json)?\s*/i, '')
      .replace(/\s*```\s*$/i, '')
      .trim()
    try {
      return JSON.parse(cleaned) as T
    } catch (e) {
      console.error('[PartyPlanner] JSON parse failed. Raw response:', raw)
      throw new Error('Party plan could not be parsed. Please try again.')
    }
  }

Use this instead of bare JSON.parse everywhere in generatePartyMenu().

### Rendering — display costs correctly:
Never render raw floats. Always format:
  const fmt = (n: number) => `$${Number(n).toFixed(2)}`
  // e.g. fmt(9.2) → "$9.20"  |  fmt(9.200000000001) → "$9.20"

Remove the canEdit field from the schema entirely — it was unused and creates confusion.

## FEATURE #5 — ARRIVAL-TIME NOTIFICATIONS

### Step 1 — Check and install
Check package.json for expo-notifications. If missing:
  npx expo install expo-notifications

### Step 2 — Store notification IDs (NOT cancel-all)
CRITICAL: Do NOT use Notifications.cancelAllScheduledNotificationsAsync().
That cancels every notification in the app including non-party ones.
Instead, store only party notification IDs:

  import AsyncStorage from '@react-native-async-storage/async-storage'
  const PARTY_NOTIF_KEY = 'partyPlannerNotifIds'

  const clearPartyNotifications = async () => {
    try {
      const raw = await AsyncStorage.getItem(PARTY_NOTIF_KEY)
      const ids: string[] = raw ? JSON.parse(raw) : []
      await Promise.all(ids.map(id => Notifications.cancelScheduledNotificationAsync(id)))
      await AsyncStorage.removeItem(PARTY_NOTIF_KEY)
    } catch { /* ignore — best effort */ }
  }

### Step 3 — Request permissions with denied-state handling
  import * as Notifications from 'expo-notifications'
  import Constants from 'expo-constants'

  const requestAndGetPushToken = async (): Promise => {
    const { status } = await Notifications.requestPermissionsAsync()
    if (status !== 'granted') {
      console.warn('[Notifications] Permission denied — skipping push token')
      return null  // graceful — in-app reminders still work
    }
    try {
      const projectId = Constants.expoConfig?.extra?.eas?.projectId
      if (!projectId) {
        console.warn('[Notifications] No EAS projectId found in app config')
        return null
      }
      const token = await Notifications.getExpoPushTokenAsync({ projectId })
      return token.data
    } catch (e) {
      console.error('[Notifications] Push token fetch failed:', e)
      return null
    }
  }

### Step 4 — Schedule party notifications
  const schedulePartyNotifications = async (
    timeline: { hoursBeforeArrival: number; task: string }[],
    arrivalTimeMs: number  // Unix timestamp in ms — use arrivalTime.getTime()
  ) => {
    await clearPartyNotifications()
    const scheduledIds: string[] = []

    for (const entry of timeline) {
      const triggerMs = arrivalTimeMs - (entry.hoursBeforeArrival * 3600 * 1000)
      if (triggerMs <= Date.now()) continue  // skip past times silently

      try {
        const id = await Notifications.scheduleNotificationAsync({
          content: {
            title: '🎉 Party prep reminder',
            body: entry.task,
          },
          trigger: { type: 'date', date: new Date(triggerMs) },
        })
        scheduledIds.push(id)
      } catch (e) {
        console.warn('[Notifications] Failed to schedule:', entry.task, e)
        // Continue — partial scheduling is better than total failure
      }
    }

    // Persist IDs so we can cancel individually later
    await AsyncStorage.setItem(PARTY_NOTIF_KEY, JSON.stringify(scheduledIds))

    // Request push token (non-blocking — don't await in main save flow)
    requestAndGetPushToken().then(token => {
      if (token) AsyncStorage.setItem('partyPlannerPushToken', token)
    })
  }

Call schedulePartyNotifications(plan.timeline, arrivalTime.getTime()) after the plan
is saved. Also call it after any regeneration that changes the timeline.

### Step 5 — In-app countdown (with cleanup)
  const [nextTask, setNextTask] = useState<{ label: string; msUntil: number } | null>(null)

  useEffect(() => {
    if (!plan || !arrivalTime) return

    const computeNext = () => {
      const now = Date.now()
      const arrivalMs = arrivalTime.getTime()
      const upcoming = plan.timeline
        .map(t => ({
          label: t.task,
          msUntil: arrivalMs - (t.hoursBeforeArrival * 3600 * 1000) - now,
        }))
        .filter(t => t.msUntil > 0)
        .sort((a, b) => a.msUntil - b.msUntil)
      setNextTask(upcoming[0] ?? null)
    }

    computeNext()
    const interval = setInterval(computeNext, 60_000)
    return () => clearInterval(interval)  // REQUIRED — prevents memory leak
  }, [plan, arrivalTime])

  // Render the countdown card if nextTask exists:
  {nextTask && (
    
      Next: {nextTask.label}
      In {Math.round(nextTask.msUntil / 60_000)} minutes
    
  )}
